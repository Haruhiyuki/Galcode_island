  三、核心数据模型

  3.1 Agent 状态机

  /// Agent 状态枚举 (与前端 types/agent.ts 一一对应)
  #[derive(Debug, Clone, Copy, PartialEq, Eq, Serialize, Deserialize)]
  pub enum AgentStatus {
      Idle,              // 空闲
      Starting,          // 正在启动 Agent 进程
      Running,           // Agent 进程已启动，等待首条 hook
      Thinking,          // LLM 推理中
      Processing,        // 正在执行工具
      WaitingApproval,   // 等待用户审批
      Completed,         // 当前任务完成
      Error,             // 发生错误
  }

  3.2 Hook 事件结构 (参考 CodeIsland 的 HookEvent)

  /// 从 Agent 输出解析得到的标准化事件
  #[derive(Debug, Clone, Serialize, Deserialize)]
  pub struct HookEvent {
      pub event_name: String,          // 标准化后的事件名: "PreToolUse", "Stop" 等
      pub session_id: Option<String>,  // 会话 ID
      pub tool_name: Option<String>,   // 工具名: Bash, Read, Edit, Write...
      pub tool_use_id: Option<String>, // 工具调用唯一 ID
      pub tool_input: Option<serde_json::Value>,  // 工具输入参数
      pub cwd: Option<String>,         // 当前工作目录
      pub model: Option<String>,       // LLM 模型名
      pub agent_id: Option<String>,    // 子 Agent ID
      pub raw_json: serde_json::Value, // 原始 JSON (保留完整信息)
  }

  3.3 Session 快照 (参考 CodeIsland 的 SessionSnapshot)

  /// 单个 Agent 会话的完整状态快照
  #[derive(Debug, Clone, Serialize, Deserialize)]
  pub struct SessionSnapshot {
      pub session_id: String,
      pub agent_type: String,          // "claude-code", "opencode", "codex"...
      pub status: AgentStatus,
      pub cwd: Option<String>,         // 项目路径
      pub model: Option<String>,       // 使用的模型
      pub current_tool: Option<String>,      // 当前工具
      pub tool_description: Option<String>,  // 工具描述
      pub tool_history: Vec<ToolHistoryEntry>, // 最近工具历史
      pub last_user_prompt: Option<String>,
      pub last_assistant_message: Option<String>,
      pub start_time: chrono::DateTime<chrono::Utc>,
      pub last_activity: chrono::DateTime<chrono::Utc>,
      pub pid: Option<u32>,            // 子进程 PID
      pub interrupted: bool,           // 是否被用户中断
  }

  /// 工具历史条目
  #[derive(Debug, Clone, Serialize, Deserialize)]
  pub struct ToolHistoryEntry {
      pub tool: String,
      pub description: Option<String>,
      pub timestamp: chrono::DateTime<chrono::Utc>,
      pub success: bool,
  }

  3.4 副作用模型 (纯 reducer 的输出)

  /// reduce_event() 返回的副作用，由调用方执行
  #[derive(Debug, Clone)]
  pub enum SideEffect {
      /// 启动进程监控
      MonitorProcess { session_id: String, pid: u32 },
      /// 停止进程监控
      StopMonitor { session_id: String },
      /// 移除会话
      RemoveSession { session_id: String },
      /// 入队完成通知 (任务结束后触发气泡)
      EnqueueCompletion { session_id: String },
      /// 推送前端事件: 状态变化
      EmitStatusChanged { session_id: String, status: AgentStatus },
      /// 推送前端事件: 工具执行
      EmitToolUpdate { session_id: String, tool: String, description: Option<String> },
      /// 推送前端事件: 任务列表更新
      EmitTodoUpdate { session_id: String, todos: Vec<TodoItem> },
      /// 推送前端事件: 日志
      EmitLog { session_id: String, level: String, message: String },
      /// 推送前端事件: 任务完成
      EmitSessionComplete { session_id: String, summary: Option<String>, emotion: Option<String> },
  }

  4.2 Rust → 前端 (Tauri Events)

  // ===== ipc/events.rs =====
  // 所有事件通过 app_handle.emit(...) 推送

  // 状态变化 (最频繁)
  app_handle.emit("agent://status-changed", StatusChangedPayload {
      session_id: "...",
      status: "Thinking",       // ← AgentStatus 的字符串形态
      tool_name: Some("Bash"),  // 可选, processing 状态时携带
      tool_description: Some("npm run build"),
  })?;

  // 工具开始执行
  app_handle.emit("agent://tool-update", ToolUpdatePayload {
      session_id: "...",
      tool: "Bash",
      description: "npm run build",
      tool_use_id: "toolu_xxx",
  })?;

  // 工具执行结果
  app_handle.emit("agent://tool-result", ToolResultPayload {
      session_id: "...",
      tool: "Bash",
      success: true,
      output: Some("Build completed successfully"),
  })?;

  // 权限审批请求
  app_handle.emit("agent://permission-request", PermissionRequestPayload {
      session_id: "...",
      tool_name: "Bash",
      tool_description: "rm -rf /path",
      tool_use_id: "toolu_xxx",
      raw_input: json!({"command": "rm -rf /path"}),
  })?;

  // 任务完成 (含 LLM 总结和情绪反馈)
  app_handle.emit("agent://session-complete", SessionCompletePayload {
      session_id: "...",
      summary: Some("修改了 3 个文件...".to_string()),
      emotion: Some("太棒了，任务全部完成！🎉".to_string()),
  })?;

  // 单条日志
  app_handle.emit("agent://log", LogPayload {
      session_id: "...",
      level: "info",
      message: "PreToolUse: Bash - npm install",
      timestamp: "2026-05-02T12:00:00Z",
  })?;

  // 错误
  app_handle.emit("agent://error", ErrorPayload {
      session_id: "...",
      message: "Agent process exited unexpectedly",
      code: "PROCESS_EXITED",
  })?;

  五、核心设计：Hook 事件解析策略

  5.1 方案选型

  针对不同 Agent 的 hook 输出方式，采用分层兼容策略：

  ┌────────┬─────────────────────┬───────────────────────────┬──────────────────────────────────────────┐
  │ 优先级 │        方案         │        适用 Agent         │                 实现方式                 │
  ├────────┼─────────────────────┼───────────────────────────┼──────────────────────────────────────────┤
  │ 首选   │ stdout JSON 行解析  │ Claude Code (已配置 hook) │ 子进程 stdout 逐行读取，反序列化 JSON    │
  ├────────┼─────────────────────┼───────────────────────────┼──────────────────────────────────────────┤
  │ 备选   │ 文件监控            │ 所有 Agent                │ notify crate 监控 Agent 的 hook 输出文件 │
  ├────────┼─────────────────────┼───────────────────────────┼──────────────────────────────────────────┤
  │ 兜底   │ stdout 文本正则匹配 │ 未配置 hook 的 Agent      │ 正则匹配工具调用模式                     │
  └────────┴─────────────────────┴───────────────────────────┴──────────────────────────────────────────┘

  5.2 Claude Code Hook 配置 (自动注入)

  在启动 Claude Code 时，通过环境变量或 CLI 参数自动注入 hook 配置：

  // ~/.claude/settings.json (由 Rust 自动写入)
  {
    "hooks": {
      "PreToolUse": [
        {
          "matcher": "*",
          "hooks": [{
            "type": "command",
            "command": "echo '{\"hook_event_name\":\"PreToolUse\",\"tool_name\":\"$CLAUDE_TOOL_NAME\",\"tool_input\":$CLAUDE_TOOL_INPUT}' >>
  /tmp/galcode_hooks.jsonl"
          }]
        }
      ]
    }
  }

  更优方案：stdout 内联输出。启动 Claude Code 时加 --print 参数（如果支持），或者通过 --append-system-prompt 注入指令让 Agent 在每次工具调用前后输出特定 JSON
  行到 stdout。Rust 侧通过 child.stdout.take() 获取异步流，逐行解析。

  5.3 事件标准化 (参考 CodeIsland 的 EventNormalizer)

  /// 事件名标准化：将不同 Agent 的异构事件名映射为统一名称
  pub fn normalize_event_name(raw: &str) -> &str {
      match raw.to_lowercase().as_str() {
          // Claude Code
          "pre_tool_use" | "pretooluse"      => "PreToolUse",
          "post_tool_use" | "posttooluse"    => "PostToolUse",
          "post_tool_use_failure"             => "PostToolUseFailure",
          "user_prompt_submit"                => "UserPromptSubmit",
          "stop" | "session_end"             => "Stop",
          "session_start"                     => "SessionStart",
          "subagent_start"                    => "SubagentStart",
          "subagent_stop"                     => "SubagentStop",
          "permission_request"                => "PermissionRequest",
          "notification"                      => "Notification",
          // OpenCode / Codex / Gemini ... (后续扩展)
          _ => raw,
      }
  }

  Backend Phase 1 — 基础 IPC + 窗口配置 (对应前端 P1)

  3. 实现 window_utils.rs：Windows 点击穿透 (移植自 xsun)
  4. 编写 ipc/commands.rs 的第一个命令：select_project_folder (调用 tauri-plugin-dialog)

  6. 更新 capabilities/default.json，增加必要的权限

  ---
  Backend Phase 2 — Agent 进程管理 (对应前端 P2)

  目标：能够从 Rust 启动和停止 Agent CLI 子进程。

  具体工作：
  1. agent/config.rs：定义各 Agent 的可执行路径和默认参数
  pub struct AgentConfig {
      pub name: String,
      pub executable: String,      // "claude" / "opencode" / "codex"
      pub args: Vec<String>,       // 默认参数
      pub env_vars: HashMap<String, String>,  // hook 配置环境变量
  }
  2. agent/launcher.rs：fn launch(config: &AgentConfig, cwd: &Path) -> Result<Child>
  3. agent/manager.rs：AgentManager 结构体
    - sessions: HashMap<String, AgentSession> (session_id → 进程句柄 + 元数据)
    - fn spawn(&mut self, config, cwd) -> String (返回 session_id)
    - fn kill(&mut self, session_id) -> Result<()>
  4. ipc/commands.rs：注册 launch_agent、stop_agent 命令
  5. 启动成功/失败后 emit 相应事件到前端

  关键数据结构：
  struct AgentSession {
      session_id: String,
      agent_type: String,
      child: tokio::process::Child,
      pid: u32,
      start_time: Instant,
      cwd: PathBuf,
  }

  验证：前端点启动 → Rust spawn 子进程 → 前端收到 agent://status-changed { status: "Starting" }。

  ---
  Backend Phase 3 — Hook 事件解析 + Session 状态机 (对应前端 P3)

  目标：这是最核心的阶段。解析 Agent 输出，驱动状态机，推送事件到前端。

  具体工作：
  1. hook/event.rs：定义 HookEvent 结构体 + fn from_json_line(line: &str) -> Result<HookEvent>
  2. hook/normalizer.rs：实现 normalize_event_name() + 事件名别名映射表
  3. hook/parser.rs：HookParser — 逐行读取 stdout，解析 JSON，推送事件
  impl HookParser {
      async fn read_loop(
          stdout: ChildStdout,
          tx: mpsc::Sender<HookEvent>,
      ) -> Result<()> {
          let reader = BufReader::new(stdout);
          let mut lines = reader.lines();
          while let Some(line) = lines.next_line().await? {
              if let Ok(event) = HookEvent::from_json_line(&line) {
                  tx.send(event).await.ok();
              }
          }
          Ok(())
      }
  }
  4. session/snapshot.rs：定义 SessionSnapshot 结构体
  5. session/state.rs：实现纯 reducer (参考 CodeIsland 的 reduceEvent)
  /// 纯函数: 给定当前会话状态和事件, 返回新状态 + 副作用列表
  pub fn reduce_event(
      snapshot: &mut SessionSnapshot,
      event: &HookEvent,
  ) -> Vec<SideEffect>;
  5. 状态转换规则：
  Idle + SessionStart   → Starting    + MonitorProcess
  Idle + UserPromptSubmit → Processing
  Processing + PreToolUse → Running   + EmitToolUpdate
  Running + PostToolUse  → Processing + EmitToolResult
  * + Stop               → Idle       + EnqueueCompletion
  * + Error              → Error      + EmitError
  6. session/cleanup.rs：tokio 定时任务，清理超时的 idle 会话 (参考 CodeIsland 的 cleanupIdleSessions)
  7. ipc/events.rs：将 SideEffect::EmitXxx 转化为 app_handle.emit(...)
  8. 将全部模块串联到 lib.rs 的 setup() 中

  验证：启动 Claude Code → 前端实时看到 status 从 Starting → Running → Processing → Idle 的变化 + 工具调用日志。

  ---
  Backend Phase 4 — 权限审批 + 任务完成 (对应前端 P4/P5)

  目标：处理需要用户确认的工具调用，任务完成后生成总结。

  具体工作：
  1. 权限审批流：
    - Rust 收到 PermissionRequest 事件 → 暂停 Agent stdin (等待用户决策)
    - emit agent://permission-request 到前端
    - 前端调用 respond_permission → Rust 向 Agent stdin 写入 allow/deny
  2. 任务完成处理：
    - Stop 事件 → SideEffect::EnqueueCompletion
    - 收集 session 的工具历史 → 构建总结文本
    - emit agent://session-complete (含 summary + emotion)

  验证：Agent 执行危险操作时前端弹出确认，任务完成后气泡展示总结。

  ---
  Backend Phase 5 — LLM 总结生成 (对应前端 P5 增强)

  目标：调用 LLM API 生成人性化的任务总结和情绪反馈。

  具体工作：
  1. llm/client.rs：
  pub async fn generate_summary(
      tool_history: &[ToolHistoryEntry],
      last_prompt: &str,
      api_config: &LlmConfig,
  ) -> Result<SummaryResult>;
    - 构建 Prompt：将工具历史 + 用户原始 prompt 组合，请求 LLM 生成 2-3 句总结 + 1 句情绪反馈
    - 调用 DeepSeek / OpenAI API
    - 返回 { summary, emotion }
  2. llm/prompt.rs：总结生成的 System Prompt 模板
  3. 在 SideEffect::EnqueueCompletion 的处理中插入 LLM 调用

  验证：任务完成后气泡展示 "修改了 3 个文件，新增了登录功能 → 你今天效率很高！继续保持~"
