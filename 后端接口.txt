  pub fn spawn_opencode_process(cfg: &AgentConfig, cwd: &Path, task_en: &str)
      -> Result<std::process::Child, String>
  {
      let mut cmd = Command::new(&cfg.executable);
      cmd.args(["run", "--format", "json", "--dir"])
         .arg(cwd)
         .arg("--dangerously-skip-permissions")  // 自动化场景跳过权限确认
         .arg(task_en)
         .current_dir(cwd)
         .stdout(Stdio::piped())
         .stderr(Stdio::inherit());
      for (k, v) in &cfg.env_vars {
          cmd.env(k, v);
      }
      cmd.spawn().map_err(|e| format!("启动 OpenCode 失败: {}", e))
  }

  1. WelcomeView:

  三、后端怎么做——启动链技术路径

  核心问题：用户选文件夹 + 选 Agent 后，后端怎么在指定文件夹启动指定 Agent

  入口：commands.rs::start_agent() 收到三个参数：

  userInputZh = "帮我写下载脚本"
  cwd         = "/home/shiju/proj"    ← 用户在 WelcomeView 选的文件夹
  agent       = "opencode"            ← 用户在 WelcomeView 选的 Agent

  后端要做的事（按调用顺序）：

  步骤 1：commands.rs — 路由 Agent 类型

  #[tauri::command]
  fn start_agent(
      app: AppHandle,
      state: State<Arc<AppState>>,
      user_input_zh: String,
      cwd: Option<String>,
      agent: Option<String>,  // 新增参数，前端已经在传了
  ) -> Result<LaunchResult, String> {
      let cwd = cwd.unwrap_or_else(|| ".".to_string());
      let agent_type = agent.unwrap_or_else(|| "opencode".to_string());

      match agent_type.as_str() {
          "opencode" => manager::launch_opencode_agent(app, state, cwd, user_input_zh),
          "demo"     => manager::launch_demo_agent(app, state, cwd, user_input_zh),
          _          => Err(format!("不支持的 Agent: {}", agent_type)),
      }
  }

  步骤 2：launcher.rs — 创建 OpenCode 子进程，工作目录设为 cwd

  pub fn spawn_opencode_process(cfg: &AgentConfig, cwd: &Path, task_en: &str)
      -> Result<std::process::Child, String>
  {
      Command::new(&cfg.executable)
          .args(["run", "--format", "json", "--dir"])
          .arg(cwd)                           // ← 关键：在用户选的文件夹里执行
          .arg("--dangerously-skip-permissions")
          .arg(task_en)                        // ← 已翻译的英文任务
          .current_dir(cwd)                    // ← Rust 进程自己也 cd 到该目录
          .stdout(Stdio::piped())              // ← stdout 被父进程捕获
          .stderr(Stdio::inherit())            // ← stderr 打印到终端
          .spawn()
  }

  实际执行的命令等价于在终端跑：
  cd /home/shiju/proj
  opencode run --format json --dir /home/shiju/proj --dangerously-skip-permissions "Write a download script"
